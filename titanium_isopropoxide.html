def read_file(file_path):  
    with open(file_path, 'rb') as f:  
        return f.read()  
  
# Чтение зашифрованных файлов  
file1_path = 'flag2.txt.enc'  
file2_path = 'flag3.txt.enc'  
enc_data2 = read_file(file1_path)  
enc_data3 = read_file(file2_path)  
  
print(enc_data2)  
print(enc_data3)


Знаем, что зашифрованные данные начинаются с tjctf{, поэтому используем это это как известный открытый текст для получения начального ключевого потока.
Напишем функцию derive_keystream, которая использует XOR между зашифрованными данными и известным открытым текстом.

def derive_keystream(enc_data, known_plaintext):
    keystream = []
    for i in range(len(known_plaintext)):
        keystream.append(enc_data[i] ^ known_plaintext[i])
    return bytes(keystream)

# Получение начального ключевого потока из обоих файлов
keystream2 = derive_keystream(enc_data2[:len(known_plaintext)], known_plaintext)
keystream3 = derive_keystream(enc_data3[:len(known_plaintext)], known_plaintext)

print(f"Начальный ключевой поток из flag2.txt.enc: {keystream2}")
print(f"Начальный ключевой поток из flag3.txt.enc: {keystream3}")


Напишем функцию decrypt_with_keystream, чтобы расшифровать данные, используя ключевой поток. Эта функция просто применяет XOR между зашифрованными данными и ключевым потоком.

def decrypt_with_keystream(enc_data, keystream):
    decrypted_data = []
    for i in range(len(enc_data)):
        decrypted_data.append(enc_data[i] ^ keystream[i % len(keystream)])
    return bytes(decrypted_data)

# Расшифровка содержимого с использованием начального ключевого потока
decrypted_flag2 = decrypt_with_keystream(enc_data2, keystream2)
decrypted_flag3 = decrypt_with_keystream(enc_data3, keystream3)

print(f"Расшифрованное содержимое из flag2.txt.enc: {decrypted_flag2}")
print(f"Расшифрованное содержимое из flag3.txt.enc: {decrypted_flag3}")

# Печать первых 60 байт для анализа
print(f"Первые 60 байт из flag2.txt.enc: {decrypted_flag2[:60]}")
print(f"Первые 60 байт из flag3.txt.enc: {decrypted_flag3[:60]}")
