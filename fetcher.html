Проанализируем код. У нас есть веб-приложение на Node.js, которое содержит ендпоинт fetch для загрузки содержимого указанного URL.

app.post('/fetch', async (req, res) => {
    const url = req.body.url;

    if (!/^https?:\/\//.test(url))
        return res.send('invalid url');

    try {
        const checkURL = new URL(url);

        if (checkURL.host.includes('localhost') || checkURL.host.includes('127.0.0.1'))
            return res.send('invalid url');
    } catch (e) {
        return res.send('invalid url');
    }

    const r = await fetch(url, { redirect: 'manual' });

    const fetched = await r.text();

    res.send(fetched);
});

Также есть ендпоинт flag для выдачи флага. Тут происходит проверка на IP-адрес обращения к поинту. Если он не локальный - сообщение bad ip.

if (req.ip !== '::ffff:127.0.0.1' && req.ip !== '::1' && req.ip !== '127.0.0.1')
        return res.send('bad ip');

Понимаем что можно отправить на fetch адрес флага и он локально к нему обратится и выдаст результат.
