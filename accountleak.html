# Определяем максимальный диапазон для перебора sub  
max_sub = 2 ** 20  # Это произвольный верхний предел для перебора; можно изменить в зависимости от времени выполнения  
  
# Пробуем возможные значения для sub  
for possible_sub in range(1, max_sub):  
    try:  
        sum_pq = (n + possible_sub ** 2 - dont_leak_this) // possible_sub  
        # Решаем квадратное уравнение x^2 - sum_pq*x + n = 0  
        discriminant = sum_pq ** 2 - 4 * n  
        if discriminant < 0:  
            continue  
  
        sqrt_discriminant = int(math.isqrt(discriminant))  
  
        # Проверяем, является ли дискриминант идеальным квадратом

            continue  
  
        p = (sum_pq + sqrt_discriminant) // 2  
        q = (sum_pq - sqrt_discriminant) // 2  
  
        # Проверяем, являются ли p и q действительно множителями n  
        if p * q == n and isPrime(p) and isPrime(q):

            phi_n = (p - 1) * (q - 1)
